	<xsd:element name="edad" type="tipoEdad"/>

Peso, decimal y valores de 0 a 1000

<xsd:schema xmlns="http://www.w3.org/2001/XMLSchema">
	<xsd:element name="peso" type="tipoPeso"/>
	<xsd:simpleType name="tipoPeso">
		<xsd:restriction base="xsd:decimal">
			<xsd:minInclusive value="0"/>
			<xsd:maxInclusive value="1000"/>
		</xsd:restriction>
	</xsd:simpleType>
</xsd:schema>


<xsd:schema xmlns="http://www.w3.org/2001/XMLSchema">
	<xsd:element name="nif" type="tipoNIF"/>
	<xsd:simpleType name="tipoNIF">
		<xsd:restriction base="xsd:string">
			<xsd:pattern value="[0-9]{8}[A-Z]"/>
		</xsd:restriction>
	</xsd:simpleType>
</xsd:schema>

	public void escritura() throws IOException{
		
		
			//creación del fichero
			File fichero1 = new File("Quijote.txt");
			
			FileWriter escribe = new FileWriter(fichero1);
			
		try {
			
			for(int i=0;i<texto.length();i++) {
				escribe.write(texto.charAt(i));
			}
			
			escribe.close();
			
		} catch (IOException e) {
			System.out.println("Error 1");
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
		
	}
	
	public void lectura() throws IOException{
		
		File fichero1 = new File("Quijote.txt");
		
		FileReader lee = new FileReader(fichero1);
		
		int c = 0;
		
		try {
		
		while(c!=-1) {
			
			c = lee.read();
			char letra = (char)c;
			System.out.print(letra);
			
		}
		
		System.out.println("\n");
		
		lee.close();
		
		}catch(IOException e) {
			System.out.println("Error 1");
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
	}
	
	public void escritura_buffer() throws IOException{
		
		try {
			
			File fl = new File("Quijote.txt");
			File fe = new File("Quijote_lineas.txt");
			
			FileReader fr = new FileReader(fl);
			FileWriter fw = new FileWriter(fe);
			
			BufferedWriter bw = new BufferedWriter(fw);
			BufferedReader br = new BufferedReader(fr);
			
			
			int caracter = 0;
			
			while(caracter!=-1) {
				
				texto = texto + (char) caracter;
				
				if((char) caracter == '.' || (char) caracter == ',' || (char) caracter == ';') {

					for(int i=0;i<texto.length();i++) {
						bw.write(texto.charAt(i));
						//escribe.write("\r\n");
					}
					bw.newLine();
					bw.flush();
					texto = "";

				}
		
			}
			
			
			while(texto!=null) {
				texto = br.readLine();
			
				if(texto!=null) {
					System.out.println(texto);
				}
			}
			
			bw.close();
			
		} catch (IOException e) {
			System.out.println("Error 3");
			// TODO Auto-generated catch block
			e.printStackTrace();
		}
		
	}
